## Q3. findOne() and find()


db.students.findOne({ city: "Chennai" })

db.students.find({ city: "Chennai" })

db.orders.findOne({ customer: "Arjun" })

db.orders.find({ customer: "Arjun" })


## Q4. Projection


db.students.find(
   {},
   { name: 1, city: 1, _id: 0 }
)

db.orders.find(
   {},
   { customer: 1, item: 1, _id: 0 }
)



## Q5. $eq


db.students.find({
   city: { $eq: "Chennai" }
})

db.orders.find({
   status: { $eq: "delivered" }
})




## Q6. $lt and $lte


db.students.find({
   gpa: { $lt: 8.0 }
})

db.students.find({
   gpa: { $lte: 8.0 }
})



## Q7. $gt and $gte


db.students.find({
   gpa: { $gt: 8.0 }
})

db.students.find({
   gpa: { $gte: 8.0 }
})



## Q8. $ne


db.students.find({
   city: { $ne: "Chennai" }
})

db.orders.find({
   status: { $ne: "delivered" }
})



## Q9. $in


db.students.find({
   city: { $in: ["Chennai", "Madurai"] }
})

db.orders.find({
   status: { $in: ["delivered", "shipped"] }
})




## Q10. $nin


db.students.find({
   city: { $nin: ["Chennai", "Madurai"] }
})

db.orders.find({
   status: { $nin: ["delivered", "shipped"] }
})



## Q11. $and



db.students.find({
   $and: [
      { city: "Chennai" },
      { gpa: { $gt: 8 } }
   ]
})

db.orders.find({
   $and: [
      { status: "delivered" },
      { price: { $gt: 5000 } }
   ]
})


## Q12. $or


db.students.find({
   $or: [
      { city: "Chennai" },
      { gpa: { $gt: 8.5 } }
   ]
})

db.orders.find({
   $or: [
      { status: "pending" },
      { price: { $gt: 50000 } }
   ]
})


## Q13. $not


db.students.find({
   gpa: { $not: { $gt: 8 } }
})

db.orders.find({
   price: { $not: { $gt: 10000 } }
})
